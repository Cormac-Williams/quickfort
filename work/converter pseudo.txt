parse blueprint:
  Dictionary<int, Blueprint> Layers;
  
  
  read all the lines in
  pull off the #mode line 
  until (eof)
    start a new layer (initially 0)
    if (!line like #,#,#...)
      height++
      layer.rows.Add(line.Split(,).ToList().Exclude("#"))
      width = maxof(width, layer.rows[latest].length)
    if got line (#>)
      start a new layer +1
      lowestDepth = lowestDepth + 1  
      layerDir = 1
    if got line (#<)
      start a new layer -1         
      highestDepth = highestDepth - 1
      layerDir = -1
